22def try_(function, item, return_item='item'): 

23 # type: (Callable[[Any], Any], Any, Any) -> Any 

24 ''' 

25 Call given function on given item, catch any exceptions and return given

27 

28 Args: 

29 function (function): Function of signature lambda x: x. 

30 item (object): Item used to call function. 

31 return_item (object, optional): Item to be returned. Default: "item". 

32 

33 Returns: 

34 object: Original item if return_item is "item". 

35 Exception: If return_item is "error".

69def relative_path(module, path): 

70 # type: (Union[str, Path], Union[str, Path]) -> Path 

71 ''' 

72 Resolve path given current module's file path and given suffix. 

73 

74 Args: 

75 module (str or Path): Always __file__ of current module. 

76 path (str or Path): Path relative to __file__. 

77 

78 Returns: 

79 Path: Resolved Path object. 

80 ''' 

81 module_root = Path(module).parent 

82 path_ = Path(path).parts # type: Any 

83 path_ = list(dropwhile(lambda x: x == ".", path_)) 

84 up = len(list(takewhile(lambda x: x == "..", path_))) 

85 path_ = Path(*path_[up:]) 

86 root = list(module_root.parents)[up - 1] 

87 output = Path(root, path_).absolute() 

88 

89 LOGGER.debug( 

90 f'relative_path called with: {module} and {path_}. Returned: {output}') 

91 return output

95def get_function_signature(function): 

96 # type: (Callable) -> Dict 

97 ''' 

98 Inspect a given function and return its arguments as a list and its keyword 

99 arguments as a dict. 

100 

101 Args: 

102 function (function): Function to be inspected. 

103 

104 Returns: 

105 dict: args and kwargs. 

106 ''' 

107 spec = inspect.getfullargspec(function) 

108 args = list(spec.args) 

109 kwargs = {} # type: Any 

110 if spec.defaults is not None: 

111 args = args[:-len(spec.defaults)] 

112 kwargs = list(spec.args)[-len(spec.defaults):] 

113 kwargs = dict(zip(kwargs, spec.defaults)) 

114 return dict(args=args, kwargs=kwargs)

117def api_function(wrapped=None, **kwargs): 

118 # type: (Optional[Callable], **Any) -> Callable 

119 ''' 

120 A decorator that enforces keyword argument only function signatures and 

121 required keyword argument values when called. 

122 

123 Args: 

124 wrapped (function): For dev use. Default: None. 

125 \*\*kwargs (dict): Keyword arguments. # noqa: W605 

126 

127 Raises: 

128 TypeError: If non-keyword argument found in functionn signature. 

129 ValueError: If keyword arg with value of '<required>' is found. 

130 

131 Returns: 

132 api function. 

133 ''' 

134 @wrapt.decorator 

135 def wrapper(wrapped, instance, args, kwargs): 

136 sig = get_function_signature(wrapped) 

137 

138 # ensure no arguments are present 

139 if len(sig['args']) > 0: 

140 msg = 'Function may only have keyword arguments. ' 

141 msg += f"Found non-keyword arguments: {sig['args']}." 

142 raise TypeError(msg) 

143 

144 # ensure all required kwarg values are present 

145 params = sig['kwargs'] 

146 params.update(kwargs) 

147 for key, val in params.items(): 

148 if val == '<required>': 

149 msg = f'Missing required keyword argument: {key}.' 

150 raise ValueError(msg) 

151 

152 LOGGER.debug(f'{wrapped} called with {params}.') 

153 return wrapped(*args, **kwargs) 

154 return wrapper(wrapped)
